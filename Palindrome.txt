public class Solution 
{
           
    
   public bool IsPalindrome(int x) 
   {
       int n,sum=0,t;
    
      t=x;
        while(x>0){
                   n=x%10;      
           sum=(sum*10)+n;      
           x=x/10; 
                  }
            
            if(t == sum)
            {
                return true;
            }
            else
            {
                return false;
            }
   }
        
    
}
